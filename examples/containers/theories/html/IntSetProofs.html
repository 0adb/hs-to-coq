The IntSet formalization

This module contains a formalization of Haskell's Data.IntSet which implements a set of integers as a patricia trie.

Most significant differing bit

Only properly defined if both arguments are non-negative.

Definition msDiffBit : Z → Z → N :=
  fun n m ⇒ Z.to_N (Z.succ (Z.log2 (Z.lxor n m))).

Lemma msDiffBit_sym: ∀ p1 p2,
  msDiffBit p1 p2 = msDiffBit p2 p1.

Section msDiffBit.
  Variable p1 p2 : Z.
  Variable (Hnonneg1 : 0 ≤ p1).
  Variable (Hnonneg2 : 0 ≤ p2).
  Variable (Hne : p1 ≠ p2).


  Lemma msDiffBit_Different:
        Z.testbit p1 (Z.pred (Z.of_N (msDiffBit p1 p2)))
     ≠ Z.testbit p2 (Z.pred (Z.of_N (msDiffBit p1 p2))).

  Lemma msDiffBit_Same:
    ∀ j, Z.of_N (msDiffBit p1 p2) ≤ j →
    Z.testbit p1 j = Z.testbit p2 j.

  Lemma msDiffBit_shiftr_same:
        Z.shiftr p1 (Z.of_N (msDiffBit p1 p2))
     = Z.shiftr p2 (Z.of_N (msDiffBit p1 p2)).
End msDiffBit.

Lemma msDiffBit_less:
  ∀ z1 z2 b,
    z1 ≠ z2 →
    Z.shiftr z1 (Z.of_N b) = Z.shiftr z2 (Z.of_N b) →
    (msDiffBit z1 z2 ≤ b)%N.

Dyadic intervals

A dyadic interval is a set of the form 
   [a⋅2^n,…,(a+1)⋅2^n-1)] where a∈Z,n≥0
which can be described by the prefix p and the shift width a. In the folloing, we just say “range” for a dyadic interval.

Definition range := (Z × N)%type.
Definition rPrefix : range → Z := fun '(p,b) ⇒ Z.shiftl p (Z.of_N b).
Definition rBits : range → N := snd.

Lemma rPrefix_rBits_range_eq:
  ∀ r1 r2, rPrefix r1 = rPrefix r2 → rBits r1 = rBits r2 → r1 = r2.

Operation: inRange

This operation checks if a value is in the range of the range set.

Definition inRange : Z → range → bool :=
  fun n '(p,b) ⇒ Z.shiftr n (Z.of_N b) =? p.

Lemma rPrefix_inRange:
  ∀ r, inRange (rPrefix r) r = true.

Lemma bit_diff_not_in_range:
  ∀ r j i,
    Z.of_N (rBits r) ≤ j →
    Z.testbit (rPrefix r) j ≠ Z.testbit i j →
    inRange i r = false.

Operation: isSubrange


Definition isSubrange : range → range → bool :=
  fun r1 r2 ⇒ inRange (rPrefix r1) r2 && (rBits r1 <=? rBits r2)%N.

Operation: rangeDisjoint

Range sets have the nice property that they are either contained in each other, or they are completely disjoint.

Definition rangeDisjoint : range → range → bool :=
  fun r1 r2 ⇒ negb (isSubrange r1 r2 || isSubrange r2 r1).

Lemma rangeDisjoint_sym: ∀ r1 r2,
  rangeDisjoint r1 r2 = rangeDisjoint r2 r1.

Lemma different_prefix_same_bits_disjoint:
  ∀ r1 r2,
    rPrefix r1 ≠ rPrefix r2 → rBits r1 = rBits r2 → rangeDisjoint r1 r2 = true.

Lemma inRange_both_not_disj:
  ∀ i r1 r2,
  inRange i r1 = true →
  inRange i r2 = true →
  rangeDisjoint r1 r2 = false.

Lemma rangeDisjoint_inRange_false:
  ∀ i r1 r2, rangeDisjoint r1 r2 = true → inRange i r1 = true → inRange i r2 = false.

Lemma disjoint_rPrefix_differ:
  ∀ r1 r2,
    rangeDisjoint r1 r2 = true → rPrefix r1 ≠ rPrefix r2.

Lemma isSubrange_disj_disj_r:
  ∀ r1 r2 r3,
  isSubrange r2 r3 = true →
  rangeDisjoint r1 r3 = true →
  rangeDisjoint r1 r2 = true.

Lemma isSubrange_disj_disj_l:
  ∀ r1 r2 r3,
  isSubrange r1 r2 = true →
  rangeDisjoint r2 r3 = true →
  rangeDisjoint r1 r3 = true.

Lemma smaller_not_subrange_disjoint:
  ∀ r1 r2,

  isSubrange r1 r2 = false →

Operation: halfRange

Non-singelton sets can be partitioned into two halfs.

Definition halfRange : range → bool → range :=
  fun '(p,b) h ⇒
    let b' := N.pred b in
    let p' := Z.shiftl p 1 in
    (if h then Z.lor p' 1 else p', b').

Lemma isSubrange_halfRange:
  ∀ r h,
    (0 < rBits r)%N →
    isSubrange (halfRange r h) r = true.

Lemma testbit_halfRange_false:
 ∀ r i h,
    (0 < rBits r)%N →
    inRange i r = true →
    inRange i (halfRange r h) = false ↔ Z.testbit i (Z.pred (Z.of_N (rBits r))) = negb h.

Lemma testbit_halfRange_true_false:
 ∀ r i,
    (0 < rBits r)%N →
    inRange i r = true →
    inRange i (halfRange r true) = false ↔ Z.testbit i (Z.pred (Z.of_N (rBits r))) = false.

Lemma testbit_halfRange_true:
 ∀ r i,
    (0 < rBits r)%N →
    inRange i r = true →
    inRange i (halfRange r true) = Z.testbit i (Z.pred (Z.of_N (rBits r))).

Lemma halfRange_inRange_testbit:
 ∀ r i h,
    (0 < rBits r)%N →
    inRange i r = true →
    inRange i (halfRange r h) = negb (xorb h (Z.testbit i (Z.pred (Z.of_N (rBits r))))).

Lemma rBits_halfRange:
  ∀ r h, rBits (halfRange r h) = N.pred (rBits r).

Lemma halfRange_isSubrange_testbit:
  ∀ r1 r2 h,
   (rBits r1 < rBits r2)%N →
   isSubrange r1 r2 = true →
   isSubrange r1 (halfRange r2 h) = negb (xorb h (Z.testbit (rPrefix r1) (Z.pred (Z.of_N (rBits r2))))).

Lemma testbit_halfRange_isSubrange:
  ∀ r1 r2,
    (rBits r1 < rBits r2)%N →
    isSubrange r1 r2 = true →
    isSubrange r1 (halfRange r2 true) = Z.testbit (rPrefix r1) (Z.pred (Z.of_N (rBits r2))).

Lemma testbit_halfRange_false_false:
  ∀ r i,
    (0 < rBits r)%N →
    inRange i r = true →
    inRange i (halfRange r false) = false ↔ Z.testbit i (Z.pred (Z.of_N (rBits r))) = true.

Lemma testbit_halfRange:
  ∀ r h,
  (0 < rBits r)%N →
  Z.testbit (rPrefix (halfRange r h)) (Z.pred (Z.of_N (rBits r))) = h.

Lemma halves_disj_aux_aux:
  ∀ r h1 h2,
  (0 < rBits r)%N →
  h2 = negb h1 →
  isSubrange (halfRange r h1) (halfRange r h2) = false.

Lemma halves_disj_aux:
  ∀ r,
  (0 < rBits r)%N →
  rangeDisjoint (halfRange r false) (halfRange r true) = true.

Lemma halves_disj:
  ∀ r1 r2 r,
  (0 < rBits r)%N →
  isSubrange r1 (halfRange r false) = true →
  isSubrange r2 (halfRange r true) = true →
  rangeDisjoint r1 r2 = true.

Lemma smaller_subRange_other_half :
  ∀ r1 r2,
    (rBits r1 < rBits r2)%N →
    isSubrange r1 r2 = true →
    isSubrange r1 (halfRange r2 true) = negb (isSubrange r1 (halfRange r2 false)).

Lemma halfRange_smaller:
  ∀ r h, (0 < rBits r)%N → (rBits (halfRange r h) < rBits r)%N.

Operation: rNonneg

This predicate inciates that the range covers non-negative numbers. We often have to restrict ourselves to these as negative numbers have an infinite number of bits set, which means that msDiffBit would not work.
If we would switch to a finite signed type, this could be dropped.

Definition rNonneg : range → Prop :=

    rNonneg r1 ↔ rNonneg r2.

  ∀ r h,
    (0 < rBits r)%N →
    rNonneg (halfRange r h) ↔ rNonneg r.

Range-related tactics

This heavily backracking tactic solves goals of the form inRange i r = false, by exploring all isSubrange r r2 = true assumptions, as well as some known lemmas about isSubrange.

Ltac inRange_false :=
  assumption ||
  multimatch goal with
    | [ H : isSubrange ?r1 ?r2 = true
        |- inRange ?i ?r1 = false ] ⇒
        apply (inRange_isSubrange_false i r1 r2 H)
    | [ |- inRange ?i (halfRange ?r ?b) = false ] ⇒
        apply (inRange_isSubrange_false i (halfRange r b) r);
          [apply isSubrange_halfRange; assumption |]
    | [ Hdis : rangeDisjoint ?r1 ?r2 = true , H : inRange ?i ?r1 = true
        |- inRange ?i ?r2 = false ] ⇒
        apply (rangeDisjoint_inRange_false i r1 r2 Hdis H)
    | [ Hdis : rangeDisjoint ?r2 ?r1 = true , H : inRange ?i ?r1 = true
        |- inRange ?i ?r2 = false ] ⇒
        rewrite rangeDisjoint_sym in Hdis;
        apply (rangeDisjoint_inRange_false i r1 r2 Hdis H)
    | [ H: inRange ?i (halfRange ?r false) = true
        |- inRange ?i (halfRange ?r true) = false ] ⇒
        eapply rangeDisjoint_inRange_false;
        [apply halves_disj_aux; auto | apply H]
    | [ H: inRange ?i (halfRange ?r true) = true
        |- inRange ?i (halfRange ?r false) = false ] ⇒
        eapply rangeDisjoint_inRange_false;
        [rewrite rangeDisjoint_sym; apply halves_disj_aux; auto | apply H]
  end; try inRange_false.

rMask

Calculates a mask in the sense of the IntSet implementation: A single bit set just to the right of the prefix. (Somewhat illdefined for singleton ranges).

Definition rMask : range → Z :=
   fun '(p,b) ⇒ 2^(Z.pred (Z.of_N b)).

Lemmas about nomatch, zero and their combinations


Lemma nomatch_spec:
  ∀ i r,
  (0 < rBits r)%N →
  nomatch i (rPrefix r) (rMask r) =
  negb (inRange i r).

Lemma zero_spec:
  ∀ i r,
  (0 < rBits r)%N →
  zero i (rMask r) = negb (Z.testbit i (Z.pred (Z.of_N (rBits r)))).

The IntSet code has a repeating pattern consisting of calls to nomatch and zero. The following two lemmas capture that pattern concisely.

Lemma nomatch_zero:
  ∀ {a} i r (P : a → Prop) left right otherwise,
  (0 < rBits r)%N →
  (inRange i r = false → P otherwise) →
  (inRange i (halfRange r false) = true → inRange i (halfRange r true) = false → P left) →
  (inRange i (halfRange r false) = false → inRange i (halfRange r true) = true → P right) →
  P (if nomatch i (rPrefix r) (rMask r) then otherwise else
     if zero i (rMask r) then left else right).

Well-formed IntSets.

This section introduces the predicate to describe the well-formedness of an IntSet. It has parameters that describe the range that this set covers, and a function that carries it denotation. This way, invariant preservation and functional correctness of an operation can be expressed in one go.

Inductive Desc : IntSet → range → (Z → bool) → Prop :=
  | DescTip : ∀ p bm r f,
    0 ≤ p →
    p = rPrefix r →
    rBits r = N.log2 WIDTH →
    (∀ i, f i = bitmapInRange r bm i) →
    isBitMask bm →
    Desc (Tip p bm) r f
  | DescBin : ∀ s1 r1 f1 s2 r2 f2 p msk r f,
    Desc s1 r1 f1 →
    Desc s2 r2 f2 →
    (0 < rBits r)%N →
    isSubrange r1 (halfRange r false) = true →
    isSubrange r2 (halfRange r true) = true →
    p = rPrefix r →
    msk = rMask r →
    (∀ i, f i = f1 i || f2 i) →
    Desc (Bin p msk s1 s2) r f.

A variant that also allows Nil, or sets that do not cover the full given range, but are certainly contained in them. This is used to describe operations that may delete elements.

Inductive Desc0 : IntSet → range → (Z → bool) → Prop :=
  | Desc0Nil : ∀ r f, (∀ i, f i = false) → Desc0 Nil r f
  | Desc0NotNil :
      ∀ s r f r' f',
      ∀ (HD : Desc s r f),
      ∀ (Hsubrange: isSubrange r r' = true)
      (Hf : ∀ i, f' i = f i),
      Desc0 s r' f'.

A variant that also allows Nil and does not reqiure a range. Used for the top-level specification.

Inductive Sem : IntSet → (Z → bool) → Prop :=
  | SemNil : ∀ f, (∀ i, f i = false) → Sem Nil f
  | DescSem : ∀ s r f (HD : Desc s r f), Sem s f.
